\begin{frame}
\frametitle{Statistics and Probability}
\Large
\[P(X<0.8) = \int^{0.8}_{0} \frac{3}{2}x^2 + x dx   \]

\[ P(X<0.8) = \left[ \frac{3x^3}{4}  + \frac{x^2}{2}\right]^{0.8}_{0}  \]

\[ P(X<0.8) = (0.384 + 0.32) - (0+0) = 0.704 \]

\[P(X>0.8) =  1 - P(X<0.8) = 1-0.704 = 0.296 \]

\end{frame}
